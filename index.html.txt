// DARK MODE
const toggle = document.getElementById('dark-mode-toggle');
toggle.addEventListener('click', e=>{
e.preventDefault();
document.body.classList.toggle('dark');
});

// NAVIGATION
document.getElementById('nav-articles').addEventListener('click',()=>document.getElementById('articles').scrollIntoView({behavior:'smooth'}));
document.getElementById('nav-projets').addEventListener('click',()=>document.getElementById('projets').scrollIntoView({behavior:'smooth'}));
document.getElementById('nav-contact').addEventListener('click',()=>document.getElementById('contact').scrollIntoView({behavior:'smooth'}));

// ARTICLE SEARCH
const searchInput = document.getElementById('search-bar');
searchInput.addEventListener('keyup',()=>{
const filter = searchInput.value.toLowerCase();
document.querySelectorAll('.article-card').forEach(card=>{
const title = card.dataset.title.toLowerCase();
card.style.display = title.includes(filter)?'block':'none';
});
});

// SHOW/HIDE ARTICLES
document.querySelectorAll('.article-card[data-article]').forEach(card=>{
card.addEventListener('click',()=>{
const articleId = card.dataset.article;
document.getElementById(articleId).style.display='block';
window.scrollTo({top:0,behavior:'smooth'});
});
});
document.querySelectorAll('.article-content .close-btn').forEach(btn=>{
btn.addEventListener('click',()=>{
btn.parentElement.style.display='none';
});
});
</script>
